  <script>
    // Year
    document.getElementById('year').textContent = new Date().getFullYear();

    // Theme toggle
    const toggle = document.getElementById('themeToggle');
    toggle.addEventListener('click', () => {
      const c = document.documentElement.classList;
      const dark = c.toggle('dark');
      localStorage.setItem('theme', dark ? 'dark' : 'light');
    });

    // Scroll progress bar
    const progress = document.getElementById('progress');
    const scrollProgress = () => {
      const h = document.documentElement;
      const scrolled = (h.scrollTop) / (h.scrollHeight - h.clientHeight);
      progress.style.width = (scrolled * 100) + '%';
    };
    document.addEventListener('scroll', scrollProgress, { passive: true });
    scrollProgress();

    // Section reveal + nav highlight
    const observer = new IntersectionObserver((entries)=>{
      entries.forEach(e=>{
        if(e.isIntersecting){
          e.target.classList.add('show');
          const id = e.target.id;
          document.querySelectorAll('a[data-nav]').forEach(a=>{
            a.classList.toggle('text-violet-600', a.getAttribute('data-nav')===id);
          });
        }
      });
    }, { threshold:.2 });
    document.querySelectorAll('.reveal').forEach(el=>observer.observe(el));

    // Tiny 3D tilt
    const tilts = document.querySelectorAll('.tilt');
    tilts.forEach((el)=>{
      const r = 10; // max tilt deg
      el.addEventListener('mousemove', (e)=>{
        const b = el.getBoundingClientRect();
        const x = (e.clientX - b.left)/b.width - .5;
        const y = (e.clientY - b.top)/b.height - .5;
        el.style.transform = `perspective(800px) rotateX(${(-y*r).toFixed(2)}deg) rotateY(${(x*r).toFixed(2)}deg)`;
      });
      el.addEventListener('mouseleave', ()=>{ el.style.transform=''; });
    });
  </script>
